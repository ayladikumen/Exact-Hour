# Exact Hour - Python dependencies (Release 1: clock + phone control)
# =============================================================================
# Install on the Raspberry Pi, inside the project's virtual environment:
#     python3 -m venv venv
#     source venv/bin/activate
#     pip install -r requirements.txt
#
# See basic_installation.md for the full setup walkthrough.
# (The phone-control server, remote_control.py, uses only the standard library.)
# =============================================================================

luma.led_matrix      # drives the MAX7219 LED matrix over SPI (pulls in luma.core + Pillow)
gpiozero             # simple, debounced reading of the UP / DOWN / START buttons
lgpio                # GPIO backend gpiozero uses on Raspberry Pi OS (Bookworm)
spidev               # SPI backend luma needs to reach the MAX7219
